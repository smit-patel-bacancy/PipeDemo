Pipes with Paramters

  • {{currentDate | date:'short'}}
  • {{currentDate | date:'shortTime'}}
  • {{currentDate | date:'fullDate'}}
  • {{currentDate | date:'shortDate'}}
  • {{currentDate | date: 'd/M/y' }}
  • {{currentDate | date: 'dd/MM/yyyy' }}
  • {{currentDate | date: 'dd-MM-yyyy' }}
  • {{currentDate | date: 'yyyy-MM/dd' }}

  • {{ 1234.56 | currency:'INR' }}
  • {{ 1234.56 | currency:'INR':'code' }}
  • {{ 1234.56 | currency:'INR':'symbol' }}

  • {{ 3.14159265 | number: '3.1-2' }}
  • {{ 3.14159265 | number: '1.4-4' }}

  • {{ 0.123456 | percent }}
  • {{ 0.123456 | percent: '2.1-2' }}
  • {{ 0.123456 | percent : "10.4-4" }}

  • {{ [1,2,3,4,5,6] | slice:1:3 }}
  • {{ [1,2,3,4,5,6] | slice:2 }}
  • {{ [1,2,3,4,5,6] | slice:2:-1 }}
